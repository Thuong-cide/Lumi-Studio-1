## PHẦN 2 — RATE LIMITING (chống spam & bảo vệ server)

### 2.1 Tạo file mới: `src/lib/rate-limit.ts`

```typescript
// Rate limiting in-memory — phù hợp cho single-instance Replit
// Không cần Redis, không cần thư viện ngoài

interface RateLimitEntry {
  count:     number
  resetTime: number
}

const store = new Map<string, RateLimitEntry>()

// Tự dọn entries cũ mỗi 5 phút để không leak memory
setInterval(() => {
  const now = Date.now()
  for (const [key, entry] of store.entries()) {
    if (entry.resetTime < now) store.delete(key)
  }
}, 5 * 60 * 1000)

export interface RateLimitResult {
  success:   boolean
  remaining: number
  resetTime: number
}

export function rateLimit(
  key: string,
  options: { windowMs: number; max: number }
): RateLimitResult {
  const now   = Date.now()
  const entry = store.get(key)

  if (!entry || entry.resetTime < now) {
    store.set(key, { count: 1, resetTime: now + options.windowMs })
    return { success: true, remaining: options.max - 1, resetTime: now + options.windowMs }
  }

  if (entry.count >= options.max) {
    return { success: false, remaining: 0, resetTime: entry.resetTime }
  }

  entry.count++
  return { success: true, remaining: options.max - entry.count, resetTime: entry.resetTime }
}

export function getClientIp(req: Request): string {
  return (
    req.headers.get('x-forwarded-for')?.split(',')[0].trim() ??
    req.headers.get('x-real-ip') ??
    'unknown'
  )
}
```

### 2.2 Áp dụng vào `src/app/api/public/album/[slug]/select/route.ts`

Thêm 10 dòng vào đầu hàm `POST`, ngay sau dòng `try {`. Giữ nguyên toàn bộ code bên dưới:

```typescript
import { rateLimit, getClientIp } from '@/lib/rate-limit'

export async function POST(req: NextRequest, { params }: { params: Promise<{ slug: string }> }) {
  // ── THÊM MỚI: Rate limit ──
  const ip    = getClientIp(req)
  const limit = rateLimit(`select:${ip}`, { windowMs: 60_000, max: 60 })
  if (!limit.success) {
    return NextResponse.json(
      { error: 'Quá nhiều yêu cầu. Vui lòng chờ một chút rồi thử lại.' },
      {
        status: 429,
        headers: { 'Retry-After': String(Math.ceil((limit.resetTime - Date.now()) / 1000)) },
      }
    )
  }
  // ── KẾT THÚC THÊM MỚI ──

  // ... giữ nguyên toàn bộ code hiện có bên dưới ...
}
```

## PHẦN 3 — HTTP CACHE CHO DRIVE PROXY

**File:** `src/app/api/drive/proxy/[fileId]/route.ts`

Chỉ thay đổi phần `headers` trong `return new NextResponse(...)`. Giữ nguyên toàn bộ logic stream:

```typescript
// TRƯỚC:
return new NextResponse(webStream, {
  headers: {
    'Content-Type': photo.mimeType || 'image/jpeg',
    'Cache-Control': 'public, max-age=3600, stale-while-revalidate=86400',
  },
})

// SAU — tăng cache lên 24h, thêm ETag:
return new NextResponse(webStream, {
  headers: {
    'Content-Type': photo.mimeType || 'image/jpeg',
    // Ảnh trên Drive không đổi sau khi upload → cache 24h
    'Cache-Control': 'public, max-age=86400, stale-while-revalidate=604800',
    // ETag giúp browser không re-download nếu ảnh chưa thay đổi
    'ETag': `"${fileId}"`,
  },
})
```

## PHẦN 4 — PRISMA CONNECTION POOL

**File:** `src/lib/prisma.ts`

Thay toàn bộ file bằng:

```typescript
import { PrismaClient } from '@prisma/client'

const globalForPrisma = globalThis as unknown as { prisma: PrismaClient }

export const prisma =
  globalForPrisma.prisma ??
  new PrismaClient({
    log: process.env.NODE_ENV === 'development' ? ['error'] : ['error'],
    datasources: {
      db: {
        url: process.env.DATABASE_URL,
      },
    },
  })

if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = prisma

// Đóng connection gracefully khi process kết thúc
process.on('beforeExit', async () => {
  await prisma.$disconnect()
})
```

Thêm vào `.env` (hoặc `.env.example`):

```env
# Giới hạn connection pool — phù hợp Replit với PostgreSQL hosted
# Tăng nếu dùng PlanetScale/Supabase, giữ thấp nếu dùng free tier
DATABASE_URL="postgresql://...?connection_limit=10&pool_timeout=20"
```

> ⚠️ Thêm `?connection_limit=10&pool_timeout=20` vào cuối DATABASE_URL hiện có nếu chưa có. Điều chỉnh `connection_limit` theo giới hạn của database provider:
> - Supabase free: tối đa 60 connections → dùng `connection_limit=10`
> - Neon free: tối đa 100 → dùng `connection_limit=15`
> - PlanetScale: serverless, không cần giới hạn

## PHẦN 5 — CACHE METADATA ALBUM PHÍA PUBLIC

**File:** `src/app/api/public/album/[slug]/route.ts`

Thêm Cache-Control header vào response. Chỉ thay đổi dòng `return NextResponse.json(album)`:

```typescript
// TRƯỚC:
return NextResponse.json(album)

// SAU — cache metadata 60 giây:
return NextResponse.json(album, {
  headers: {
    // Metadata album thay đổi ít — cache 60s, stale 5 phút
    'Cache-Control': 'public, s-maxage=60, stale-while-revalidate=300',
  },
})
```
